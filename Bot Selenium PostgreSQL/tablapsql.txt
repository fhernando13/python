CREATE DATABASE ssd;
USE ssd;

drop table if exists ssd_table;
CREATE TABLE IF NOT EXISTS ssd_table(
	id serial not null,
	description varchar(255),
	vendedor varchar(100),
	precio integer,
      descuento varchar (50),
	meses integer,
	pagos integer,
      envio varchar (30),
	primary key(id)
);
